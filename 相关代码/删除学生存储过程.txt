create or replace procedure s_delete(v_sid in number,vresult out number) 
as
vcnt number:=0;
begin
   select count(*) into vcnt from stu_inf where sid=v_sid;
     if vcnt= 1 then
       delete from stu_inf where sid=v_sid;
         vresult:=1;
      else
         vresult:=0;   
      end if;
    commit;
    exception
      when others then
           vresult:=2;
end s_delete;
